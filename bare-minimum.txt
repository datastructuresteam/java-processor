// Program to calculate sum of two numbers.
public class TestProgram
{
public static void main(String[] args)
{
int n1,n2;
n1=5;
n2=10
int answer=0;
answer=n1+n2;
for (int i = 0; i < 10; i++)
{
if (i % 2 ==0)
{
System.out.println(answer);
}
else
{
System.out.println();
}
}// end of for loop
}// end of main method
}// end of TestProgram
